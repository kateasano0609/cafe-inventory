        async function loadData() {

            document.getElementById('loading').style.display = 'block';

            document.getElementById('content').style.display = 'none';

            document.getElementById('error').style.display = 'none';

            

            try {

                const url = `https://docs.google.com/spreadsheets/d/${SHEET_ID}/gviz/tq?tqx=out:csv&sheet=${SHEET_NAME}`;

                const response = await fetch(url);

                const csvText = await response.text();

                

                // Parse CSV data

                const rows = csvText.split('\n').map(row => {

                    // Simple CSV parser that handles quoted fields

                    const result = [];

                    let current = '';

                    let inQuotes = false;

                    

                    for (let i = 0; i < row.length; i++) {

                        const char = row[i];

                        if (char === '"') {

                            inQuotes = !inQuotes;

                        } else if (char === ',' && !inQuotes) {

                            result.push(current.trim());

                            current = '';

                        } else {

                            current += char;

                        }

                    }

                    result.push(current.trim());

                    return result;

                });

                

                // Convert to inventory data (skip header row)

                inventoryData = rows.slice(1).filter(row => row.length >= 6 && row[0]).map(row => ({

                    category: row[0] || '',

                    item: row[1] || '',

                    vendor: row[2] || '',

                    par: parseFloat(row[3]) || 0,

                    onHand: parseFloat(row[4]) || 0,

                    waste: parseFloat(row[5]) || 0

                }));

                

                document.getElementById('loading').style.display = 'none';

                document.getElementById('content').style.display = 'block';

                renderInventory();

                

            } catch (error) {

                console.error('Error loading data:', error);

                document.getElementById('loading').style.display = 'none';

                document.getElementById('error').style.display = 'block';

            }

        }

                    categories[item.category] = [];

                }

                categories[item.category].push(item);

            });

            

            let totalItems = 0;

                `;

                                       min="0" 

                                       onchange="updateItem(${itemIndex}, 'waste', this.value)"

                                       ontouchstart="this.focus()">

                            </div>
